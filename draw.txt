int xPos=200;
int xDir=1;
void setup()
{
  size(1000,1000);
}
void draw()
{
  background(0);
  ellipse(xPos, 200, 20, 20);
  xPos=xPos+xDir;
  if (xPos>width || xPos<10)
  {
    xDir=-xDir;
  }
  ellipse(xPos, 400, 20, 20);
  xPos=xPos+xDir;
  if (xPos>width || xPos<10)
  {
    xDir=-xDir;
  }
  ellipse(xPos, 600, 20, 20);
  xPos=xPos+xDir;
  if (xPos>width || xPos<10)
  {
    xDir=-xDir;
  }
  ellipse(xPos, 800, 20, 20);
  xPos=xPos+xDir;
  if (xPos>width || xPos<10)
  {
    xDir=-xDir;
  }
}
